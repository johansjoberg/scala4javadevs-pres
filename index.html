For comprehensions

val names = List("Abraham", "Anna", "Ella", "Tomas", "Stellan")

//Declarative
val namesLongerThanFourChars = for(
  name <- names
  if name.length > 4
) yield name

println(namesLongerThanFourChars.mkString(", "))

//Imperative
var namesLongerThanFourChars = List[String]()
for(name <- names){
  if(name.length > 4) {
    // :: operator means append name to namesLongerThanFourChars
    // and create a new list
    namesLongerThanFourChars = name :: namesLongerThanFourChars
  }
}

println(namesLongerThanFourChars.mkString(", "  ))

  def apply(val firstName:String = "John", val lastName: String = "Doe") = {
    new Name(firstName, lastName)

Function

Basics

val names = List("Adam", "Stellan", "Emma", "Li")

//Defines a function and stores it in value countLength, the
//type String => Int is actually not needed as it can be inferred
val countLength: String => Int = (input: String) => input.length()

//Goes through the list and applies the function on each element
//creating a new list
val nameLength = for(name <- names) yield countLength(name)

//You can do the same using the collection method map in scala
val nameLength = names.map(countLength)

//defines a method that does the same as the previously declared function
def countLength(input: String) = input.length()

//countLength method get converted to a function at compile time,
//if it is used as a function
val nameLength = names.map(countLength)

//map using anonymous function with wildcard syntax '_'
val nameLength = names.map(_.length())

//map using anonymous function using syntax with type inferred
val nameLength = names.map( name => name.length())

import collection.GenSeq

//class for simulating work
case class Worker(workInMs: Long) {
  def work() = {
    Thread.sleep(workInMs)
    "Done!"
  }
}

//method for measuring the time of the execution
def measureTime(workers: List[Worker])(work: GenSeq[Worker] => GenSeq[String]) = {
  val start = System.currentTimeMillis()
  work(workers)
  val stop = System.currentTimeMillis()
  stop - start
}

//create a list of 100 Worker objects
val workers = List.fill(100)(Worker(100))

//sequential work
val sequentialTime = measureTime(workers) {
  _.map(_.work())
}

//parallel work
val parallelTime = measureTime(workers) {
  _.par.map(_.work())
}
