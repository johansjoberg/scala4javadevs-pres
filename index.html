For comprehension

For comprehension

val names = List("Abraham", "Anna", "Ella", "Tomas", "Stellan")

//Similar to java for-loop
var namesLongerThanFourChars = List[String]()
for(name <- names){
  if(name.length > 4) {
    // :: operator means append name to namesLongerThanFourChars
    // and create a new list
    namesLongerThanFourChars = name :: namesLongerThanFourChars
  }
}

println(namesLongerThanFourChars.mkString(", "  ))

//Preferred way in Scala
val namesLongerThanFourChars = for(
  name <- names
  if name.length > 4
) yield name

println(namesLongerThanFourChars.mkString(", "))

Coding style

Declarative vs Imperative

Imperative example

case class Client(name: String, accounts:List[Int])

val clients = List(
  Client("Eric", List(1000, 400, 50)),
  Client("Michel", List(-10, 10, 0)),
  Client("Roger", List(0, 0, 0))
)
var clientsWithZeroBalance = List[Client]()
for(client <- clients){
  var sum = 0
  for(account <- client.accounts) {
    sum += account
  }
  if(sum == 0){
    clientsWithZeroBalance = client :: clientsWithZeroBalance
  }
}

println(clientsWithZeroBalance.mkString("\n"))

object Name { //Companion object
  def apply(firstName: String = "John", lastName: String = "Doe") = {
    new Name(firstName, lastName)
  }

val robSmith = Name("Rob", "Smith")
val annSmith = Name(lastName = "Smith", firstName = "Ann")

Function

Basics

val names = List("Adam", "Stellan", "Emma", "Li")

//Defines a function and stores it in value countLength, the
//type String => Int is actually not needed as it can be inferred
val countLength: String => Int = (input: String) => input.length()

//Goes through the list and applies the function on each element
//creating a new list
val nameLength = for(name <- names) yield countLength(name)

//You can do the same using the collection method map in scala
val nameLength = names.map(countLength)

//defines a method that does the same as the previously declared
//function
def countLength(input: String) = input.length()

//countLength method get converted to a function at compile time,
//if it is used as a function
val nameLength = names.map(countLength)

//map using anonymous function with wildcard syntax '_'
val nameLength = names.map(_.length())

//map using anonymous function using syntax with type inferred
val nameLength = names.map( name => name.length())

Function

Basics

Syntax for a one parameter function type is e.g. String => Int

A method can be used as a function

Function types can also be inferred

Higher-Order Functions

Function that take one or more functions as input parameters or a function that give a function as output.

Easier to re-use code by using functions as building blocks

Higher-Order Functions

//higher-order function that filters Strings, how they are filtered
//is decided by the predicate function input
val filter = (predicate: String => Boolean, strings: List[String]) => {
  for (
    string <- strings
    if predicate(string)
  ) yield string
}

//function of type String => Boolean
val notEmptyString = (str:String) => !str.isEmpty

val words = List("Hi", "", "there")

val realWords = filter(notEmptyString, words)
//realWords = List(”Hi”, ”There”)

Collections

Basics

import collection.GenSeq

case class Worker(workInMs: Long) { //class for simulating work
  def work() = {
    Thread.sleep(workInMs)
    "Done!"
  }
}

//method for measuring the time of the execution
def measureTime(workers: List[Worker])
               (work: GenSeq[Worker] => GenSeq[String]) = {
  val start = System.currentTimeMillis()
  work(workers)
  val stop = System.currentTimeMillis()
  stop - start
}

val workers = List.fill(100)(Worker(100)) //create 100 Workers

val sequentialTime = measureTime(workers){
  _.map(_.work())
}
val parallelTime = measureTime(workers){
  _.par.map(_.work())
}

Variable/Value pattern matching

def whichMedal(position: Int) = position match {
 case 1        => "Gold"
 case 2        => "Silver"
 case 3        => "Bronze"
 case loserPos => "You don't get medals for position " + loserPos
}

print(whichMedal(4)) //prints 'You don't get medals for position 4'

Pattern Matching

Type pattern matching

trait HasWater
case class Human extends HasWater
case class Cucumber extends HasWater

//match on types
def waterAmount(hasWater: HasWater) = hasWater match {
  case Human    => 0.65
  case Cucumber => 0.96
  case _        => throw new Exception("Don't know..")
}
println(waterAmount(new Human)) //prints '0.65'

Pattern Matching

Deconstruction pattern matching

case class Person(firstName: String, lastName: String, age: Int)
val persons = List(
  Person("Johan", "Sjöberg", 33),
  Person("Henrik", "Andersson", 32),
  Person("Anna", "Palm", 15)
)

//matcher is a function of type Person => String
val matcher = (person: Person) => person match {
  case p@Person("Johan", "Sjöberg", 33) => {
    p.firstName + " is " + p.age + " years young."
  }
  case Person(fname, _, age) if age < 18 => {
    fname + " is " + age + " years young."
  }
  case Person(fname, _, age) => {
    fname + " is " + age + " years old."
  }
}

//transform all persons to a list of strings
val output = persons.map(matcher).mkString("\n")
print(output)
